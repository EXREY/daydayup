  ```bash
  astyle \
  --mode=c \
  --style=kr \
  --indent=force-tab \
  --attach-namespaces \
  --attach-classes \
  --attach-inlines \
  --indent-switches \
  --indent-col1-comments \
  --pad-oper \
  --pad-header \
  --unpad-paren \
  --align-pointer=type \
  --fill-empty-lines \
  --add-brackets \
  --recursive \
  --suffix=none *.c *.cpp *.h *.hpp
  ```
  </details>
